2.10  price_history
CREATE TABLE price_history
CREATE TABLE price_history (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  product_id  UUID NOT NULL REFERENCES products(id) ON DELETE CASCADE,
  old_price   NUMERIC(10,2) NOT NULL,
  new_price   NUMERIC(10,2) NOT NULL,
  changed_by  UUID REFERENCES users(id),
  changed_at  TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_price_history_product ON price_history(product_id);
CREATE INDEX idx_price_history_date    ON price_history(changed_at);

2.11  eod_entries
CREATE TABLE eod_entries
CREATE TABLE eod_entries (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  entry_date  DATE NOT NULL,               -- the day these sales occurred
  product_id  UUID REFERENCES products(id) ON DELETE SET NULL,
  qty_sold    INTEGER NOT NULL,
  created_by  UUID REFERENCES users(id),
  created_at  TIMESTAMPTZ DEFAULT now()
);
-- Each row = one product's daily diary total
-- stock_log rows with reason='eod_entry' are generated on confirmation
CREATE INDEX idx_eod_date ON eod_entries(entry_date);

2.12  token_blocklist
CREATE TABLE token_blocklist
CREATE TABLE token_blocklist (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  token_jti   TEXT UNIQUE NOT NULL,  -- JWT 'jti' claim (unique token ID)
  user_id     UUID REFERENCES users(id) ON DELETE CASCADE,
  expires_at  TIMESTAMPTZ NOT NULL,  -- same as JWT exp — for cleanup
  created_at  TIMESTAMPTZ DEFAULT now()
);
-- Cleanup job: DELETE FROM token_blocklist WHERE expires_at < now();
-- Run nightly via pg_cron or a scheduled Railway job
CREATE INDEX idx_blocklist_jti ON token_blocklist(token_jti);

2.13  sessions
CREATE TABLE sessions
CREATE TABLE sessions (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id      UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  token_jti    TEXT UNIQUE NOT NULL,
  device_hint  TEXT,   -- e.g. 'Chrome on Windows' (from User-Agent)
  ip_address   TEXT,
  last_seen    TIMESTAMPTZ DEFAULT now(),
  created_at   TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_sessions_user ON sessions(user_id);

2.14  otp_requests
CREATE TABLE otp_requests
CREATE TABLE otp_requests (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email       TEXT NOT NULL,
  otp_hash    TEXT NOT NULL,   -- bcrypt hash of the 6-digit OTP
  used        BOOLEAN DEFAULT false,
  expires_at  TIMESTAMPTZ NOT NULL,  -- now() + 10 minutes
  created_at  TIMESTAMPTZ DEFAULT now()
);


3. Security Architecture — Full Suite
3.1  Rate Limiting & Account Lockout
middleware/rateLimiter.js
const rateLimit = require('express-rate-limit');

// Applied only to POST /auth/login
const loginLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,  // 15 minutes
  max: 5,                     // 5 attempts per IP per window
  message: { error: 'Too many login attempts. Try again in 15 minutes.' },
  standardHeaders: true,
  legacyHeaders: false,
});

// Account lockout stored in DB (separate from IP rate limit)
// After 5 failed attempts: users.locked_until = now() + 30 minutes
// Login check: if locked_until > now() → reject with time remaining

3.2  JWT with JTI + Revocation
services/auth.service.js — Token Generation
const { v4: uuidv4 } = require('uuid');
const jwt = require('jsonwebtoken');

function generateToken(user) {
  const jti = uuidv4();  // unique ID per token
  const token = jwt.sign(
    { sub: user.id, role: user.role, jti },
    process.env.JWT_SECRET,
    { expiresIn: '8h' }
  );
  // Record session in DB
  await db.sessions.insert({ user_id: user.id, token_jti: jti, ... });
  return token;
}

// On logout: INSERT into token_blocklist, DELETE from sessions
// On auth middleware: check token_blocklist — reject if found
// On password change: DELETE all sessions for user, INSERT all JTIs to blocklist

3.3  Password Reset — OTP Flow
Password Reset Flow — Step by Step
•  1. Staff clicks 'Forgot Password' on login screen — enters their email
•  2. Server checks email exists in users table — if not, return generic message (no user enumeration)
•  3. Generate 6-digit OTP — hash it with bcrypt, store in otp_requests with 10-minute expiry
•  4. Send OTP to email via Nodemailer (Gmail SMTP) — plain text email
•  5. Staff enters OTP on the reset screen within 10 minutes
•  6. Server: find latest unused otp_request for email, verify OTP hash, check not expired
•  7. If valid: mark OTP as used, allow staff to set new password
•  8. New password hashed with bcrypt (rounds=12), saved to users table
•  9. All active sessions for this user are immediately invalidated (all JTIs blocklisted)
•  10. Staff is redirected to login with new password
•  OWNER SHORTCUT: Owner can set any staff member's password directly from Settings — no OTP needed

3.5  Offline Price Sync on Reconnect
store/connectivity.js — Reconnect Handler
// Runs when navigator.onLine changes from false → true
async function handleReconnect() {
  // 1. Re-fetch all product prices from API
  const freshPrices = await api.products.getPrices();

  // 2. Compare against cached prices in Zustand store
  const changedProducts = freshPrices.filter(p =>
    p.price !== cachedPrices[p.id] || p.price_updated_at > lastOnlineTime
  );

  // 3. If any prices changed: show warning banner
  if (changedProducts.length > 0) {
    showPriceChangeWarning(changedProducts);
    // Banner: 'Prices changed for: Rice, Wheat Oil — please verify before billing'
  }

  // 4. Update Zustand cache with fresh prices
  updateCachedPrices(freshPrices);

  // 5. Flush offline bill sync queue
  await syncOfflineBills();
}

5. Key Business Logic
5.1  Price Change (Single Product)
PUT /products/:id — When price field changes
•  1. Read current price from products table
•  2. INSERT into price_history: { product_id, old_price, new_price, changed_by, changed_at }
•  3. UPDATE products SET price = new_price, price_updated_at = now()
•  4. Return updated product
•  Past bills are NEVER touched — bill_items.unit_price is a snapshot from time of sale

5.2  Bulk Price Update
POST /products/bulk-price — Atomic batch update
•  Request body: [{ id: 'uuid', price: 45.00 }, { id: 'uuid2', price: 120.00 }, ...]
•  Server runs the entire batch inside a single Supabase transaction
•  For each product with a changed price: INSERT price_history, UPDATE products
•  Products where new price equals current price are skipped — no log entry
•  If any single update fails: entire batch rolls back — no partial updates
•  Response: { updated: N, skipped: M, products: [...] }

5.3  EOD Stock Entry Confirmation
POST /inventory/eod — Batch stock deduction
•  Request body: { entry_date: 'YYYY-MM-DD', items: [{ product_id, qty_sold }] }
•  For each item: check product exists and is_active = true
•  Deduct: UPDATE products SET stock_qty = stock_qty - qty_sold (allow negative — audit will correct)
•  INSERT eod_entries row per item
•  INSERT stock_log row per item with reason='eod_entry' and entry_date
•  Return: total items processed, total estimated value, any products now below threshold
•  No bill number generated — this is a stock operation, not a sales transaction

5.4  Bill Confirmation — Price Snapshot
POST /bills/:id/confirm — Price integrity
•  For each bill_item: read product.price AT THE MOMENT OF CONFIRMATION
•  Store this as bill_items.unit_price — permanently fixed
•  If product price changed since item was added to cart: use the NEW current price
•  Frontend shows price age badge — operator sees if a price is stale before confirming
•  Once confirmed: changing the product price later NEVER alters the bill
•  This guarantees bills are always an accurate receipt of what was charged
